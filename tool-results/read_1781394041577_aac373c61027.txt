    20→-- ============================================================
    21→-- TABLE: stores
    22→-- ============================================================
    23→CREATE TABLE stores (
    24→    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    25→    name VARCHAR(255) NOT NULL,
    26→    slug VARCHAR(100) UNIQUE NOT NULL,
    27→    address TEXT NOT NULL,
    28→    latitude DECIMAL(10, 8) NOT NULL,
    29→    longitude DECIMAL(11, 8) NOT NULL,
    30→    location GEOGRAPHY(POINT, 4326) GENERATED ALWAYS AS (
    31→        ST_SetSRID(ST_MakePoint(longitude, latitude), 4326)::geography
    32→    ) STORED,
    33→    phone VARCHAR(20),
    34→    email VARCHAR(255),
    35→    base_delivery_fee DECIMAL(10, 2) DEFAULT 3.50,
    36→    per_km_charge DECIMAL(10, 2) DEFAULT 0.50,
    37→    free_delivery_threshold DECIMAL(10, 2) DEFAULT 20.00,
    38→    delivery_radius_km DECIMAL(5, 2) DEFAULT 5.00,
    39→    is_active BOOLEAN DEFAULT true,
    40→    created_at TIMESTAMPTZ DEFAULT now(),
    41→    updated_at TIMESTAMPTZ DEFAULT now()
    42→);
    43→
    44→-- GiST index for geospatial queries (radius-based store lookup)
    45→CREATE INDEX idx_stores_location ON stores USING GIST (location);
    46→
    47→-- ============================================================
    48→-- TABLE: profiles
    49→-- Linked to Supabase Auth users
    50→-- ============================================================
    51→CREATE TABLE profiles (
    52→    id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
    53→    store_id UUID REFERENCES stores(id) ON DELETE RESTRICT,
    54→    email VARCHAR(255) NOT NULL,
    55→    full_name VARCHAR(255) NOT NULL,
    56→    phone VARCHAR(20),
    57→    role VARCHAR(20) NOT NULL CHECK (role IN ('customer', 'owner', 'manager', 'picker', 'rider')),
    58→    avatar_url TEXT,
    59→    is_active BOOLEAN DEFAULT true,
    60→    created_at TIMESTAMPTZ DEFAULT now(),
    61→    updated_at TIMESTAMPTZ DEFAULT now()
    62→);
    63→
    64→-- Index for quick role lookups
    65→CREATE INDEX idx_profiles_store ON profiles (store_id);
    66→CREATE INDEX idx_profiles_role ON profiles (store_id, role);
    67→
    68→-- ============================================================
    69→-- TABLE: categories
    70→-- Supports nested categories via parent_id
    71→-- ============================================================
    72→CREATE TABLE categories (
    73→    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    74→    store_id UUID NOT NULL REFERENCES stores(id) ON DELETE CASCADE,
    75→    name VARCHAR(100) NOT NULL,
    76→    slug VARCHAR(100) NOT NULL,
    77→    description TEXT,
    78→    image_url TEXT,
    79→    parent_id UUID REFERENCES categories(id) ON DELETE SET NULL,
    80→    sort_order INTEGER DEFAULT 0,
    81→    is_active BOOLEAN DEFAULT true,
    82→    created_at TIMESTAMPTZ DEFAULT now()
    83→);
    84→
    85→-- Unique slug per store
    86→CREATE UNIQUE INDEX idx_categories_slug ON categories (store_id, slug);
    87→CREATE INDEX idx_categories_parent ON categories (store_id, parent_id);
    88→
    89→-- ============================================================
    90→-- TABLE: products
    91→-- Includes HFSS flag, VAT rate, and stock tracking
    92→-- ============================================================
    93→CREATE TABLE products (
    94→    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    95→    store_id UUID NOT NULL REFERENCES stores(id) ON DELETE CASCADE,
    96→    category_id UUID NOT NULL REFERENCES categories(id) ON DELETE RESTRICT,
    97→    name VARCHAR(255) NOT NULL,
    98→    slug VARCHAR(255) NOT NULL,
    99→    description TEXT,
   100→    price DECIMAL(10, 2) NOT NULL,
   101→    vat_rate DECIMAL(5, 4) NOT NULL CHECK (vat_rate IN (0.0000, 0.0500, 0.2000)),
   102→    is_hfss BOOLEAN DEFAULT false,
   103→    image_url TEXT,
   104→    barcode VARCHAR(50),
   105→    unit VARCHAR(20) DEFAULT 'each',
   106→    weight_kg DECIMAL(8, 3),
   107→    is_available BOOLEAN DEFAULT true,
   108→    stock_quantity INTEGER DEFAULT 0,
   109→    is_featured BOOLEAN DEFAULT false,
   110→    sort_order INTEGER DEFAULT 0,
   111→    created_at TIMESTAMPTZ DEFAULT now(),
   112→    updated_at TIMESTAMPTZ DEFAULT now()
   113→);
   114→
   115→-- Unique slug per store
   116→CREATE UNIQUE INDEX idx_products_slug ON products (store_id, slug);
   117→
   118→-- B-tree for category browsing per store
   119→CREATE INDEX idx_products_store_category ON products (store_id, category_id);
   120→
   121→-- GIN trigram for full-text product search
   122→CREATE INDEX idx_products_name_trgm ON products USING GIN (name gin_trgm_ops);
   123→
   124→-- B-tree for HFSS filtering in promotions
   125→CREATE INDEX idx_products_hfss ON products (is_hfss, is_featured) WHERE is_available = true;
   126→
   127→-- ============================================================
   128→-- TABLE: addresses
   129→-- Delivery addresses with PostGIS geocoding
   130→-- ============================================================
   131→CREATE TABLE addresses (
   132→    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
   133→    user_id UUID NOT NULL REFERENCES profiles(id) ON DELETE CASCADE,
   134→    label VARCHAR(50),
   135→    address_line_1 VARCHAR(255) NOT NULL,
   136→    address_line_2 VARCHAR(255),
   137→    city VARCHAR(100) NOT NULL,
   138→    postcode VARCHAR(10) NOT NULL,
   139→    latitude DECIMAL(10, 8),
   140→    longitude DECIMAL(11, 8),
   141→    location GEOGRAPHY(POINT, 4326) GENERATED ALWAYS AS (
   142→        CASE
   143→            WHEN latitude IS NOT NULL AND longitude IS NOT NULL
   144→            THEN ST_SetSRID(ST_MakePoint(longitude, latitude), 4326)::geography
   145→            ELSE NULL
   146→        END
   147→    ) STORED,
   148→    is_default BOOLEAN DEFAULT false,
   149→    created_at TIMESTAMPTZ DEFAULT now()
   150→);
   151→
   152→-- GiST index for delivery zone validation
   153→CREATE INDEX idx_addresses_location ON addresses USING GIST (location);
   154→CREATE INDEX idx_addresses_user ON addresses (user_id);
   155→
   156→-- ============================================================
   157→-- TABLE: orders
   158→-- Full order lifecycle tracking
   159→-- ============================================================
   160→CREATE TABLE orders (
   161→    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
   162→    store_id UUID NOT NULL REFERENCES stores(id) ON DELETE RESTRICT,
   163→    customer_id UUID NOT NULL REFERENCES profiles(id),
   164→    picker_id UUID REFERENCES profiles(id),
   165→    rider_id UUID REFERENCES profiles(id),
   166→    address_id UUID NOT NULL REFERENCES addresses(id) ON DELETE RESTRICT,
   167→    status VARCHAR(20) NOT NULL CHECK (status IN ('placed', 'picking', 'ready', 'out_for_delivery', 'delivered', 'cancelled')),
   168→    subtotal DECIMAL(10, 2) NOT NULL,
   169→    vat_amount DECIMAL(10, 2) NOT NULL,
   170→    delivery_fee DECIMAL(10, 2) NOT NULL,
   171→    total DECIMAL(10, 2) NOT NULL,
   172→    stripe_session_id VARCHAR(255),
   173→    stripe_payment_intent_id VARCHAR(255),
   174→    payment_status VARCHAR(20) DEFAULT 'pending' CHECK (payment_status IN ('pending', 'paid', 'failed', 'refunded')),
   175→    delivery_slot TIMESTAMPTZ,
   176→    notes TEXT,
   177→    created_at TIMESTAMPTZ DEFAULT now(),
   178→    updated_at TIMESTAMPTZ DEFAULT now()
   179→);
   180→
   181→-- B-tree for customer order history
   182→CREATE INDEX idx_orders_customer ON orders (customer_id, created_at DESC);
   183→-- B-tree for admin order management
   184→CREATE INDEX idx_orders_store_status ON orders (store_id, status);
   185→-- B-tree for rider delivery queue
   186→CREATE INDEX idx_orders_rider ON orders (rider_id, status) WHERE rider_id IS NOT NULL;
   187→
   188→-- ============================================================
   189→-- TABLE: order_items
   190→-- Line items with substitute tracking
   191→-- ============================================================
   192→CREATE TABLE order_items (
   193→    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
   194→    order_id UUID NOT NULL REFERENCES orders(id) ON DELETE CASCADE,
   195→    product_id UUID NOT NULL REFERENCES products(id),
   196→    product_name VARCHAR(255) NOT NULL,
   197→    quantity INTEGER NOT NULL CHECK (quantity > 0),
   198→    unit_price DECIMAL(10, 2) NOT NULL,
   199→    vat_rate DECIMAL(5, 4) NOT NULL,
   200→    vat_amount DECIMAL(10, 2) NOT NULL,
   201→    subtotal DECIMAL(10, 2) NOT NULL,
   202→    substitute_preference VARCHAR(20) CHECK (substitute_preference IN ('closest_match', 'do_not_substitute')),
   203→    substituted_with UUID REFERENCES products(id),
   204→    picked BOOLEAN DEFAULT false
   205→);
   206→
   207→-- B-tree for order detail retrieval

   210→-- ============================================================

   229→-- ============================================================
   230→-- HELPER: Function to update updated_at timestamp
   231→-- ============================================================
   232→CREATE OR REPLACE FUNCTION update_updated_at()
   233→RETURNS TRIGGER AS $$
   234→BEGIN
   235→    NEW.updated_at = now();
   236→    RETURN NEW;
   237→END;
   238→$$ LANGUAGE plpgsql;
   239→
   240→-- Apply updated_at triggers
   241→CREATE TRIGGER update_stores_updated_at BEFORE UPDATE ON stores
   242→    FOR EACH ROW EXECUTE FUNCTION update_updated_at();
   243→CREATE TRIGGER update_profiles_updated_at BEFORE UPDATE ON profiles
   244→    FOR EACH ROW EXECUTE FUNCTION update_updated_at();
   245→CREATE TRIGGER update_products_updated_at BEFORE UPDATE ON products
   246→    FOR EACH ROW EXECUTE FUNCTION update_updated_at();
   247→CREATE TRIGGER update_orders_updated_at BEFORE UPDATE ON orders
   248→    FOR EACH ROW EXECUTE FUNCTION update_updated_at();

   250→-- ============================================================
   251→-- HELPER: Function to get user role (for RLS policies)
   252→-- ============================================================
   253→CREATE OR REPLACE FUNCTION get_user_role()
   254→RETURNS VARCHAR AS $$
   255→    SELECT role FROM profiles WHERE id = auth.uid();
   256→$$ LANGUAGE sql SECURITY DEFINER STABLE;
   257→
   258→-- ============================================================
   259→-- HELPER: Function to get user's store_id (for RLS policies)
   260→-- ============================================================
   261→CREATE OR REPLACE FUNCTION get_user_store_id()
   262→RETURNS UUID AS $$
   263→    SELECT store_id FROM profiles WHERE id = auth.uid();
   264→$$ LANGUAGE sql SECURITY DEFINER STABLE;

   634→-- Auto-creates profile when new user signs up via Supabase Auth
   635→-- ============================================================
   636→
   637→-- Function to handle new user signup
   638→-- Creates a profile row automatically when a user registers
   639→-- IMPORTANT: If no active store exists, the profile insert is skipped
   640→-- to avoid FK violation. The profile will be created when a store
   641→-- is seeded and the user can be manually linked.
   642→CREATE OR REPLACE FUNCTION handle_new_user()
   643→RETURNS TRIGGER AS $$
   644→DECLARE
   645→    default_store_id UUID;
   646→BEGIN
   647→    -- Get the first active store as default (for single-store MVP)
   648→    SELECT id INTO default_store_id FROM stores WHERE is_active = true LIMIT 1;
   649→
   650→    -- Only insert profile if an active store exists
   651→    -- Without a valid store_id, the FK constraint on profiles.store_id
   652→    -- would reject the insert and roll back the entire auth.users INSERT,
   653→    -- making registration silently fail with no error shown to the user.
   654→    IF default_store_id IS NOT NULL THEN
   655→        INSERT INTO profiles (id, store_id, email, full_name, role)
   656→        VALUES (
   657→            NEW.id,
   658→            default_store_id,
   659→            COALESCE(NEW.email, ''),
   660→            COALESCE(NEW.raw_user_meta_data->>'full_name', 'New Customer'),
   661→            'customer'
   662→        );
   663→    ELSE
   664→        -- No store exists yet — insert profile with NULL store_id
   665→        -- This requires store_id to be nullable
   666→        INSERT INTO profiles (id, store_id, email, full_name, role)
   667→        VALUES (
   668→            NEW.id,
   669→            NULL,
   670→            COALESCE(NEW.email, ''),
   671→            COALESCE(NEW.raw_user_meta_data->>'full_name', 'New Customer'),
   672→            'customer'
   673→        );
   674→    END IF;
   675→
   676→    RETURN NEW;
   677→END;
